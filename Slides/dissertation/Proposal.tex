\begin{frame}{Motivation}

\begin{itemize}

\item Changes-in-Changes (CiC) is a nonlinear extension of Difference-in-Differences (DiD) for evaluating the effects of policy interventions or treatments (Athey and Imbens, 2006)
\vspace{0.02cm}
\item CiC targets the counterfactual outcome distributions, not just average effects.
\vspace{0.02cm}

\item CiC is designed for \textbf{binary treatment}
\vspace{0.02cm}
\item Many empirical settings involve \textbf{continuous treatment}
\vspace{0.02cm}
\item \textbf{Example}: Marginal propensity to consume (MPC) \\
  Changes in disposable income\ $\to$\ Non-durable consumption
  \vspace{0.02cm}
  \item \textbf{ Question}: Can we extend CiC to settings with continuous treatment?



\end{itemize}

\end{frame}

\begin{frame}{Motivation}
    \begin{itemize}
        \item D’Haultfoeuille, Hoderlein, and Sasaki (2023) extend Changes-in-Changes to continuous treatment in repeated cross section data
   \vspace{0.1cm}
   \item Limitation: Only certain treatment effects on the treated are identified \\
       E.g., for individuals with treatment $d$, the effects of moving from $d$ to a specific $d^\prime$ identified

         \vspace{0.1cm}
    
    \item Similar issues arise in panel data literature, e.g., Chernozhukov et al. (2013)
     \vspace{0.1cm}

\item Possible solutions:\\
     (1) Shape restrictions + three periods $\rightarrow$ partial identification\\
   (2) Linear random coefficient model $\rightarrow$  identification of average treatment effects only
   
 \vspace{0.1cm}
        \item   \textbf{Goal of this paper}: To point identify richer treatment effects with only two periods.
    \end{itemize}
\end{frame}

\begin{frame}{Setup and Notation: Panel Data}

\begin{itemize}
     \item Unit are drawn from an infinite superpopulation and observed over two periods $t\in \{0,1\}$
     \vspace{0.1cm}
     \item Each unit can be an individual, firm, or geographical unit (county or region).   
          \vspace{0.1cm}

     \item Let $D_{it}\in\mathcal{D}$ denote a continuous treatments at period $t$, and $\mathbf{D}_i=(D_{i0},D_{i1})\in \mathcal{D}_{01}$
    \\ 
    % Let $S_i=\mathds{1}\{D_{i0}\neq D_{i1}\}$ be an indicator equal to 1 if the unit's treatment changes over two periods
    \vspace{0.1cm}
    \item Let $Y_{it}(\mathbf{d}_i)$ denote potential outcomes at period $t$ under treatment path $\mathbf{d}_i=(d_{i0},d_{i1})$ 
    
    Static effect: $Y_{it}(\mathbf{d}_i)=Y_{it}(d_{it})$, $\ \ \forall \mathbf{d}_i=(d_{i0},d_{i1})$ and $t\in \{0,1\}$
    
    Consistency: observed outcome satisfies $Y_{it}=Y_{it}(D_{it})$
    \vspace{0.1cm}

    \item Suppress the individual index $i$ for simplicity when there is no ambiguity

\end{itemize}



\end{frame}



\begin{frame}{Parameter of Interest}
     Interested in $F_{Y_{t}(d)}$ for $d\in\mathcal{D}$ and its functionals , e.g.,
    
    \vspace{0.3cm}

\begin{itemize}
    \item Average structural functions $ASF_{t}(d)\coloneqq\mathbb{E}[Y_{t}(d)]$\\ 
    \vspace{0.2cm}

    Average treatment effects $ATE_t(d,d^{\prime})\coloneqq ASF_{t}(d)-ASF_{t}(d^{\prime})$ for all $d,d^{\prime}\in\mathcal{D}$

\vspace{0.2cm}
\item Quantile structural functions: $QSF_{t}(d,\tau)\coloneqq Q_{Y_{t}(d)}(\tau)$\\
\vspace{0.2cm}

Quantile treatment effects: $QTE(\tau,d,d^{\prime})\coloneqq QSF_{t}(d,\tau)-QSF_{t}(d^{\prime},\tau)$ for all $d,d^{\prime}\in\mathcal{D}$
\end{itemize} 

\end{frame}

\begin{frame}{Model and Assumptions}
    \begin{block}{Assumption 1 (Rank Invariance on the Time Trend)}\label{ass:rankinvariance}
        $$Y_{it}(d)=g_{t}\big(\tilde{Y}_{it}(d)\big)$$ and $\tilde{Y}_{it}(d)\in\mathbb{R}$, and $y\mapsto g_{t}(y)$ is strictly increasing.
    \end{block}
    \vspace{0.6cm}
\begin{itemize}
    \item $\tilde{Y}_{it}(d)$: Detrended outcome \\
    $\tilde{Y}_{it}(d)= h(d,U_{it})$ where $U_{it}$ is multidimensional unobserved heterogeneity, e.g., preference, ability
    \item $g_t(y)$: Time effects/shocks, e.g., macro shocks \\ 
    \vspace{0.1cm}
    Time effect could be nonlinear, and it preserves the ranking of individuals in outcome distribution
     % \item Athey and Imbens (2006): untreated potential outcome $Y_{t}(0)=g_{t}(U_{t})$
\end{itemize}

\end{frame}

\begin{frame}{Model and Assumptions}
\begin{block}{Assumption 2 (Time Homogeneity)}\label{ass:stationary}
  For all $d$ and $\mathbf{d}$,
    $$\tilde{Y}_{i0}(d)\mid \mathbf{D}=\mathbf{d} \ \sim \  \tilde{Y}_{i1}(d)\mid \mathbf{D}=\mathbf{d} $$

\end{block}
\vspace{0.5cm}
\begin{itemize}
    \item Individuals with the same treatment path have the same distribution of $\tilde{Y}_{it}(d)$ across time
    % \item Unobserved heterogeneity in the treatment equation is restricted to be one-dimensional
    \item Suppose $\tilde{Y}_{it}(d)=h(d,U_{it})$,
    \begin{itemize}\setlength{\leftskip}{-1.2em}
        \item[--]  A2 is equivalent to $U_{i0}\mid \mathbf{D}_i=\mathbf{d} \ \sim \ U_{i1}\mid \mathbf{D}_i=\mathbf{d}$ 
        \item[--] Sufficient condition: $U_{it}=A_i+V_{it}$ where $V_{i0}\mid A_i, \mathbf{D}_i=\mathbf{d} \sim V_{i1}\mid A_i, \mathbf{D}_i=\mathbf{d}$
    \end{itemize}
    % \item Athey and Imbens (2006): $U_{0}\mid D=d\sim U_{1}\mid D=d$ for $d\in\{0,1\}$

\end{itemize}

\end{frame}

\begin{frame}{Model and Assumptions}
\begin{block}{Assumption 3 (Quasi Stayers)}
 There exists a positive mass of individuals whose change in treatment is small, i.e., $ \mathbb{P}(\Delta D_i\leq c)>0\ \ \forall \ c >0$ 
\end{block}

\begin{block}{Assumption 4 (Regularity Conditions)}
(i) For $t\in\{0,1\}$, $F_{D_t}$ is continuous, and $F_{\tilde{Y}_{t}\mid \Delta D=0}$ is continuous and stritly increasing.\\ 
(ii) Let $m_{t}(y;d_{0},d_{1})=\mathbb{P}(\tilde{Y}_{t}\leq y\mid D_{0}=d_{0},D_{1}=d_{1})$. For each $y$, $\sup_{\vert d_{1}-d_{0}\vert<c}\vert m_{t}(y;d_{0},d_{1})-m_{t}(y;d_{0},d_{0})\vert\to0$ as $c\to0$.\\


    
\end{block}
    
\end{frame}

\begin{frame}{Identification: Time Trend $g_{t}(\cdot)$}
    \begin{itemize}
        \item Given the strict monotonicity, normalize $g_{1}(y)=y$ so that
           $$Y_{0}(d)=g_{0}(\tilde{Y}_{0}(d))\qquad  Y_{1}(d)=\tilde{Y}_{1}(d)$$
        % \begin{itemize}
            % \item[-]
        % Redefine $U_{t}(\cdot)$ as $g_{1}(U_{t}(\cdot))$ and $g_{t}(\cdot)$ as $g_{t}\left(g_{1}^{-1}(\cdot)\right)$ i.e.,
        % $Y_{0}(d)=g_{0}(U_{0}(d))$, $Y_{1}(d)=U_{1}(d)$
        % \end{itemize}        
        % \vspace{0.2cm}
        \item By using quasi-stayers, let $A_c = \{|\Delta D|< c \}$ where $\Delta D=D_1-D_0$  
        $$g_{0}(\cdot)=\lim_{c\downarrow 0 } F_{Y_{0}\mid A_c}^{-1}\left(F_{Y_{1}\mid A_c}\big(\cdot \big)\right)$$ 
        
        \item Obtain the detrended outcome $\tilde{Y}_{t}=g_{t}^{-1}(Y_{t})$ and $\tilde{Y}_{t}(d)=g_{t}^{-1}(Y_{t}(d))$ 
    \end{itemize}

\end{frame}


\begin{frame}{Modeling the Dependence Structure} \label{lemma:copula}

\begin{block}{Lemma (Sklar, 1959)}
 Given $t\in\{0,1\}$ and $d^\prime \in\mathcal{D}$, there exists a unique copula $C:[0,1]^2\to [0,1]$ such that
$$F_{\tilde{Y}_t(d^\prime),D_t}(y,d)=C_{\tilde{Y}_t(d^\prime),D_t}\Big(\ F_{\tilde{Y}_t(d^\prime)}(y)\hspace{0.1em},\ F_{D_t}(d) \hspace{0.2em} \Big)\quad \text{ for all }y\in\mathcal{Y},d\in\mathcal{D}$$
    
\end{block}

\begin{itemize}
    \item  Copula function $C_{\tilde{Y}_t(d^\prime),D_t}\big(\cdot,\cdot\big)$ captures dependence structure between $\tilde{Y}_t(d^\prime)$ and $D_t$. \ \ \hyperlink{def:copula}{\beamergotobutton{}} 
    \item Separate the marginal distributions from the (scale-free/\hspace{0.06em}rank) dependence
structure
\end{itemize}
\vspace{0.5cm}
    \begin{block}{Assumption 5 (Stability of Gaussian Copula)} 
    \begin{itemize}
        \item[(i)] Dependence between $\tilde{Y}_t(d^\prime) $ and $D_t$ is stable over time, i.e., $C_{\tilde{Y}_0(d^\prime),D_0}\big(\cdot,\cdot\big)=C_{\tilde{Y}_1(d^\prime),D_1}\big(\cdot,\cdot\big)$
    \item[(ii)] Dependence between $\tilde{Y}_t(d^\prime) $ and $D_t$ is Gaussian,
    i.e., $C_{\tilde{Y}_t(d^\prime),D_t}\big(\cdot,\cdot; \rho \big)$ is a Gaussian copula with correlation parameter $\rho$
    \end{itemize}
    
    \end{block}
\end{frame}


\begin{frame}{Discussion} 
    \begin{itemize}
        \item The restriction applies only to the dependence structure, allowing flexible marginals
        \item We can also allow non-Gaussian copulas via local Gaussian representation \label{slides:LGR} \hyperlink{lemma:LGR}{\beamergotobutton{}}
    \item Testable when more than two periods are available
        
    \end{itemize}
\end{frame}

 \end{align*}

D_{it}&=\eta_t+ \textcolor{blue}{\nu_{it}}
 \end{align*}
where $\delta_t$ and $\eta_t$ are deterministic time effects, and $\varepsilon_{it}$ could be multi-dimensional unobservables.


\end{frame}

\begin{frame}{Identification}
Assumption 5 implies that given $y$ and $d$,
\begin{align*}
    \Phi^{-1}\left(F_{\tilde{Y}_{0}\mid D_{0}}(y\mid d)\right)	&=a_{d,y}+b_{d,y}\Phi^{-1}\left(F_{D_{0}}(d)\right)\\
\Phi^{-1}\left(F_{\tilde{Y}_{1}\mid D_{1}}(y\mid d)\right)	&=a_{d,y}+b_{d,y}\Phi^{-1}\left(F_{D_{1}}(d)\right)
\end{align*}
where 
\begin{align*}
    a_{d,y}	&=\frac{\Phi^{-1}\left(F_{\tilde{Y}_1(d)}(y)\right)}{\sqrt{1-\rho_{d}^{2}}}\\
b_{d,y}	&=-\frac{\rho_{d}}{\sqrt{1-\rho_{d}^{2}}}
\end{align*}

\begin{block}{Proposition}
    Suppose $F_{D_0}(d)\neq F_{D_1}(d), \forall d\in \mathcal{D}$. Under Assumption 1 -- 5, \ $F_{Y_{1}(d)}(y)$ and $\rho_{d}$ are identified \\ for $y\in\mathcal{Y}$ and $d\in\mathcal{D}$.
\end{block} 
    
\end{frame}




\begin{frame}{Identification in Repeated Cross Sections}
\label{slides:RC}
\begin{itemize}
\item Observe two repeated cross sections $\{(Y_{i0},D_{i0})\}_{i=1}^n,\{Y_{i1},D_{i1})\}_{i=1}^n$ 
       \vspace{0.15cm}

    \item Assumption 6: Indiv. with the same treatment rank at two periods have the same dist. of $\tilde{Y}_t(d)$.\\ 
    $\tilde{Y}_{0}(d)\mid V_{0}=v \ \sim \  \tilde{Y}_{1}(d)\mid V_{1}=v$
\quad where $V_t \coloneqq F_{D_t}(D_t)$ is the ranking of $D_t$
      \vspace{0.15cm}
    \item Assumption 7: There is a crossing point of distribution of treatment in both periods.  \\ 
    There exists a $d^{*}$ s.t. $F_{D_{0}}(d^{*})=F_{D_{1}}(d^{*})\in(0,1)$  \quad \hyperlink{ass:cross}{\beamergotobutton{Illustration}}\\
    Observations with treatment level $d^{*}$ serve as a control group.
       \vspace{0.15cm}
       \item Under A7, the dependence structure between $\tilde{Y}_t(d)$ and $V_t$ is stable over time.\\ A5 (i) automatically hold.
    
\end{itemize}

\end{frame}

\begin{frame}{Introduction}
\begin{itemize}
    \item The Difference-in-Differences (DID) relies on two main assumptions
         \begin{itemize}
     \item[-] Parallel Trends (PT)
     \item[-] No Anticipation Effects (No-AE), often imposed implicitly
     \end{itemize}
     
    \item People are forward-looking, e.g. 
     \begin{itemize}
     \item[-] Healthcare utilization decreases before Medicaid expansion (Alpert, 2016)
     \item[-] Physician labor supply increases before tort reform (Malani \& Reif, 2015)
     \end{itemize}
     \item Common practice: drop a few pre-treatment periods potentially affected by anticipation. 
     \begin{itemize}
         \item We don’t always know when people start to change behavior in anticipation of a policy.
         \item Different people/groups may react at different times.
     \end{itemize}
    \item Goal: Identify the ATT while allowing for violations of the No-AE assumption
\end{itemize}
\end{frame}

\begin{frame}{Identification}
\begin{itemize}
    \item Under the Parallel Trend, the standard DID estimand identifies the ATT net of the AE:
     $$\tau^{DID} = ATT-AE$$ where $AE\equiv\mathbb{E}[Y_0(1)-Y_0(0)\mid D=1]$
     \item Propose a bounded-variation assumption (Manski and Pepper, 2018; Roth and Rambachan, 2023): for some $c\in [0,1)$, $|AE| \leq c\cdot|ATT|$ 
     

     \item With multiple pre-treatment periods, we further assume exponential c-bounded anticipation: \\for some $c \in [0,1)$, 
        $|AE_{t_0}|\leq c^{|t_0|+1}\cdot |ATT|$ for all $t_0\in \{-T_0,...,-1,0\}$\\
        (Motivated by the exponential discounting model in Malani and Reif (2015))

        \item Interpretation:
    \begin{itemize}
        \item[-] Anticipation effects are bounded and decay over time
        \item[-] Provides a flexible, nonparametric alternative to fully parametric models
        \item[-] The choice of c can play the role of sensitivity analysis
    \end{itemize}
  
\end{itemize} 

\end{frame}


\begin{frame}{Summary}
\begin{itemize}
    \item We derive bounds on the ATT that take the form of intersection-union bounds
    \item We construct a misspecification robust bound for scenarios where researchers do not know whether the PT or No-AE assumption is the one failing.
    \item We extend the framework to staggered adoption design
    \item We illustrate the practical usefulness of these bounds through two applied empirical examples.
\end{itemize}


\end{frame}

\begin{frame}{Definition of Copula} \label{def:copula}
    \begin{block}{Definition (Nelsen, 2006)} A function $C:[0,1]^2 \to [0,1]$ is a two-dimensional copula if \\
    (i) For all $u,u^\prime\in [0,1]$, and $v,v^\prime\in [0,1]$ such that $u\leq u^\prime$, and $v\leq v ^\prime$, we have 
    $$ C(u^\prime,v^\prime) + C(u,v) \geq C(u^\prime,v) + C(u,v^\prime) $$
    (ii) For all $u,v\in [0,1]$, $C(u,v)=C(0,v)=0$ and $C(1,v)=v,C(u,1)=u$
    \end{block}

\begin{frame}{Local Gaussian Representation}
\label{lemma:LGR}
    \begin{block}{Lemma (Chernozhukov et al., 2025)}
        Given $t\in\{0,1\}$ and $d^\prime\in\mathcal{D}$, the joint distribution $F_{\tilde{Y}_t(d^\prime),D_t}$ can be represented by a standard bivariate normal distribution at any point $(y,d)$ as
        $$F_{\tilde{Y}_{t}(d^{*}),D_{t}}(y,d)=\Phi_2\Big( \ 
        \ \Phi^{-1}\left(F_{\tilde{Y}_{t}(d^\prime)}(y) \right),\ \Phi^{-1}\Big( F_{D_t}(d)\Big) \ ;\ \ \textcolor{red}{\rho_{\tilde{Y}_{t}(d^\prime),D_{t}}(y,d)} \Big)$$ where $\Phi_2(\cdot,\cdot;\rho)$ is the standard bivariate normal CDF with parameter $\rho$, and $\textcolor{red}{\rho_{\tilde{Y}_{t}(d^\prime),D_{t}}(y,d)}\in[-1,1]$ is the local correlation parameter, whose value is unique.\ \ 
    \end{block}
    
    \vspace{0.4cm}
    % \pause
    \begin{itemize}
      \item $\Phi_2\Big(\Phi^{-1}(\cdot),\Phi^{-1}(\cdot),\rho\Big)$ captures the dependence structure of $\tilde{Y}_t(d^\prime)$ and $D_t$
      \item $\rho(y,d)$ measure the local dependence of $\tilde{Y}_t(d^\prime)$ and $D_t$ at $(y,d)$
      \item It is \textbf{Gaussian copula} if $\rho(y,d)=\rho$ for all $(y,d)$

    \end{itemize}

\begin{frame}{Local Gaussian Representation} \label{def:LGR}
   The joint distribution $F_{X,Y}$ can be represented by a standard bivariate normal distribution at any point $(x,y)$ as
        $$F_{X,Y}(x,y)=\Phi_2\Big( \ 
        \ \Phi^{-1}\left(F_{X}(x) \right),\ \Phi^{-1}\Big( F_{Y}(y)\Big) \ ;\ \ \rho_{X,Y}(x,y)\Big)$$ where $\Phi_2(\cdot,\cdot;\rho)$ is the standard bivariate normal CDF with parameter $\rho$, and $\rho_{X,Y}(x,y)\in[-1,1]$ is the unique solution in $\rho$ to $$F_{X,Y}(x,y)=\Phi_2\Big( \ 
        \ \Phi^{-1}\left(F_{X}(x) \right),\ \Phi^{-1}\Big( F_{Y}(y)\Big) \ ;\ \ \rho\Big)$$

\begin{frame}{Estimation}

\begin{itemize}
    \item Step 1: Estimate the crossing point $d^*$\\
$$\widehat{d}^* = \arg \min_{d\in [\widehat{F}_{D_{0}}^{-1}(\underline{p}),\widehat{F}_{D_{1}}^{-1}(\bar{p}) ]} \Big| \widehat{F}_{D_{1}}(d)-\widehat{F}_{D_{0}}(d) \Big| $$
    \item Step 2: Estimate the time trend $g_0$
    $$\widehat{g}_{0}(y)=\widehat{F}_{Y_{0}\mid D_{0}}^{-1}\left[\widehat{F}_{Y_{1}\mid D_{1}}(y\mid \widehat{d}^*)\big|\widehat{d}^*\right]$$ where
    $$\widehat{F}_{Y_{t}\mid D_{t}}(y\mid d)=\frac{\sum_{i=1}^{n}\mathds{1}\{Y_{it}\leq y\}K\left(\frac{d-D_{it}}{h_{n}}\right)}{\sum_{i=1}^{n}K\left(\frac{d-D_{it}}{h_{n}}\right)}$$ and $K(\cdot)$ a kernel function and $h_n$ is a bandwidth.\\
\vspace{0.2cm}
    Detrend in period 0:  $\tilde{Y}_0 = \widehat{g}_0^{-1}(Y_0)$
\end{itemize}
    
\end{frame}

\begin{frame}{Estimation}
    \begin{itemize}
        \item Step 3: Solve the linear system
        \begin{align*}
            \widehat{a}_{d,y}	&=\frac{\Phi^{-1}\left(\widehat{F}_{\tilde{Y}_{0}\mid D_{0}}(y\mid d)\right)\Phi^{-1}\left(\widehat{F}_{D_{1}}(d)\right)-\Phi^{-1}\left(\widehat{F}_{Y_{1}\mid D_{1}}(y\mid d)\right)\Phi^{-1}\left(\widehat{F}_{D_{0}}(d)\right)}{\Phi^{-1}\left(\widehat{F}_{D_{1}}(d)\right)-\Phi^{-1}\left(\widehat{F}_{D_{0}}(d)\right)}\\
\widehat{b}_{d,y}	&=\frac{\Phi^{-1}\left(\widehat{F}_{Y_{1}\mid D_{1}}(y\mid d)\right)-\Phi^{-1}\left(\widehat{F}_{\tilde{Y}_{0}\mid D_{0}}(y\mid d)\right)}{\Phi^{-1}\left(\widehat{F}_{D_{1}}(d)\right)-\Phi^{-1}\left(\widehat{F}_{D_{0}}(d)\right)}
        \end{align*}

\quad\qquad \widehat{\rho}_{d}(y)	=\frac{-\widehat{b}_{d,y}}{\sqrt{1+\widehat{b}_{d,y}^{2}}}$$
